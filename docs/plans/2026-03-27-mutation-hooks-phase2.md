// ── Update draft (merged before/after view) ────────────────────

/// Merged before/after view of a record being updated.
///
/// Built by the repository from `(current_record, user_patch)` before
/// invoking `before_update`. Hook authors inspect and mutate via
/// per-field [`DraftField`] accessors generated by `#[model]`.
#[derive(Debug, Clone)]
pub struct UpdateDraft<T: Clone> {
    before: T,
    after: T,
}

impl<T: Clone> UpdateDraft<T> {
    /// Create a draft where `after` starts as a clone of `before`.
    #[must_use]
    pub fn new(before: T) -> Self {
        let after = before.clone();
        Self { before, after }
    }

    /// Create a draft with explicit before and after states.
    #[must_use]
    pub fn new_with_changes(before: T, after: T) -> Self {
        Self { before, after }
    }

    /// Snapshot of the record before the mutation.
    #[must_use]
    pub fn before(&self) -> &T {
        &self.before
    }

    /// The proposed record state after applying patches and hook rewrites.
    #[must_use]
    pub fn after(&self) -> &T {
        &self.after
    }

    /// Mutable access to the proposed state (for bulk rewrites).
    #[must_use]
    pub fn after_mut(&mut self) -> &mut T {
        &mut self.after
    }

    /// Consume the draft and return the final proposed record.
    #[must_use]
    pub fn into_after(self) -> T {
        self.after
    }
}

// ── Per-field draft accessor ───────────────────────────────────

/// Borrowing per-field accessor for [`UpdateDraft`].
///
/// Borrows `before` immutably and `after` mutably from the parent draft,
/// giving hook authors the same diff helpers as [`FieldDiff`] plus the
/// ability to [`set`](DraftField::set) the proposed value.
///
/// The temporary borrow is dropped at statement end, so sequential
/// field accesses compile without borrow conflicts:
///
/// ```rust,ignore
/// if draft.status().changed_to(&Status::Approved) {
///     draft.approved_at().set(Some(now));  // new borrow, previous dropped
/// }
/// ```
pub struct DraftField<'a, T> {
    before: &'a T,
    after: &'a mut T,
}

impl<'a, T> DraftField<'a, T> {
    /// Create a new draft field accessor.
    #[must_use]
    pub fn new(before: &'a T, after: &'a mut T) -> Self {
        Self { before, after }
    }

    /// The field value before the mutation.
    #[must_use]
    pub fn before(&self) -> &T {
        self.before
    }

    /// The current proposed field value.
    #[must_use]
    pub fn after(&self) -> &T {
        self.after
    }

    /// Override the proposed field value.
    pub fn set(&mut self, value: T) {
        *self.after = value;
    }
}

impl<T: PartialEq> DraftField<'_, T> {
    /// Returns `true` if the field value differs from the original.
    #[must_use]
    pub fn changed(&self) -> bool {
        *self.before != *self.after
    }

    /// Returns `true` if the field value is identical to the original.
    #[must_use]
    pub fn unchanged(&self) -> bool {
        *self.before == *self.after
    }

    /// True if the field changed **and** the new value equals `value`.
    #[must_use]
    pub fn changed_to(&self, value: &T) -> bool {
        self.changed() && *self.after == *value
    }

    /// True if the field changed **and** the old value equals `value`.
    #[must_use]
    pub fn changed_from(&self, value: &T) -> bool {
        self.changed() && *self.before == *value
    }
}

impl<T: PartialEq> DraftField<'_, Option<T>> {
    /// True if the field went from `None` to `Some`.
    #[must_use]
    pub fn was_set(&self) -> bool {
        self.before.is_none() && self.after.is_some()
    }

    /// True if the field went from `Some` to `None`.
    #[must_use]
    pub fn was_cleared(&self) -> bool {
        self.before.is_some() && self.after.is_none()
    }
}
```

    s.split('_')

            match chars.next() {
                None => String::new(),
                Some(c) => c.to_uppercase().to_string() + &chars.collect::<String>(),

            use ::autumn_web::reexports::diesel_async::AsyncConnection;
            use ::autumn_web::reexports::diesel_async::scoped_futures::ScopedFutureExt;
            use ::autumn_web::hooks::{MutationContext, MutationOp, MutationHooks, UpdateDraft};

            let mut conn = self.pool.get().await.map_err(::autumn_web::AutumnError::from)?;
            let mut ctx = MutationContext::new(MutationOp::Update);

            // Load current record
            let current = #table_ident::table
                .find(id)
                .first::<#model_name>(&mut conn)
                .await
                .optional()
                .map_err(::autumn_web::AutumnError::from)?
                .ok_or_else(|| ::autumn_web::AutumnError::not_found_msg(
                    format!("{} with id {} not found", stringify!(#model_name), id)
                ))?;

            // Build merged draft from current + patch
            let mut draft = UpdateDraft::<#model_name>::from_patch(&current, changes);

            // before_update can inspect/rewrite via draft field accessors
            self.hooks.before_update(&mut ctx, &mut draft).await?;

            // Diff before vs after to build per-column SET clause
            let proposed = draft.into_after();

            // Transaction: persist + after_update
            let hooks = &self.hooks;
            let ctx_ref = &ctx;
            let record = conn.transaction::<#model_name, ::autumn_web::AutumnError, _>(|conn| {
                async move {
                    let record = ::autumn_web::reexports::diesel::update(#table_ident::table.find(id))
                        .set(&proposed)  // NOTE: we need the model to impl AsChangeset — see Step 4
                        .get_result::<#model_name>(conn)
                        .await
                        .map_err(::autumn_web::AutumnError::from)?;
                    hooks.after_update(ctx_ref, &record).await?;
                    Ok(record)
                }.scope_boxed()
            }).await?;

            if let Err(e) = self.hooks.after_commit(&ctx, MutationOp::Update).await {
                ::autumn_web::reexports::tracing::warn!("after_commit hook error: {e}");
            }

            Ok(record)
        };
```

**Step 4: Generate a Diesel-compatible changeset from the model**

The `proposed` value after `draft.into_after()` is the full model (e.g., `Article`). For Diesel's `.set()`, the model needs `AsChangeset`. The query struct already has `Queryable` and `Selectable` — we need to also derive `AsChangeset` on it.

In `autumn-macros/src/model.rs`, add `AsChangeset` to the query struct derives:

```rust
        #[derive(Debug, Clone, ::diesel::Queryable, ::diesel::Selectable, ::diesel::AsChangeset)]
```

This means the full model can be used as a changeset. Diesel will SET all non-PK columns, which is correct because the draft's `after` already has the right values for unchanged fields (they match the DB).

**Step 5: Update the no-hooks update path**

The no-hooks `update_body` in `autumn-macros/src/repository.rs` (~line 434-443) also needs to handle the new `Patch<T>`-based `UpdateX`. Since there's no draft in the no-hooks path, we need to convert `Patch` fields to a form Diesel can use.

Simplest approach: the no-hooks path also loads current, builds draft, diffs, and persists — it just doesn't call any hooks. This keeps the codegen simpler (one path) and avoids maintaining two separate update implementations.

Alternatively, keep the no-hooks path simpler by making it also use `from_patch` + full model set. The overhead is one extra SELECT for the current record, but it means no-hooks repos still get correct Patch<T> → SQL semantics.

#[derive(Clone, Default)]
struct SlugRewriteHooks;

impl MutationHooks for SlugRewriteHooks {
    type Model = Article;
    type NewModel = NewArticle;
    type UpdateModel = (); // unused in draft-based flow

    async fn before_update(
        &self,
        _ctx: &mut MutationContext,
        draft: &mut UpdateDraft<Article>,
    ) -> AutumnResult<()> {
        // Rewrite slug when title changes
        if draft.title().changed() {
            let new_slug = draft.title().after().to_lowercase().replace(' ', "-");
            draft.slug().set(new_slug);
        }
        Ok(())
    }
}

#[derive(Clone, Default)]
struct RejectHooks;

impl MutationHooks for RejectHooks {
    type Model = Article;
    type NewModel = NewArticle;
    type UpdateModel = ();

    async fn before_create(
        &self,
        _ctx: &mut MutationContext,
        new: &mut NewArticle,
    ) -> AutumnResult<()> {
        if new.title.is_empty() {
            return Err(autumn_web::AutumnError::bad_request_msg("title required"));
        }
        Ok(())
    }
}

    type UpdateModel = ();

        _ctx: &MutationContext,
        _record: &Article,
